20 inline constexpr int calculate_posix_open_mode(open::mode const &om)
21 {
22  using namespace open;
23  std::size_t value(remove_ate(om).value);
24  int mode(0);
25  if(value&binary.value)
26  {
27 #ifdef O_BINARY
28  mode |= O_BINARY;
29 #endif
30  value &= ~binary.value;
31  }
32  if(value&excl.value)
33  {
34  mode |= O_CREAT | O_EXCL;
35  value &= ~excl.value;
36  }
37  if(value&trunc.value)
38  {
39  mode |= O_TRUNC;
40  value &= ~trunc.value;
41  }
42  switch(value)
43  {
44 //Action if file already exists; Action if file does not exist; c-style mode; Explanation
45 //Read from start; Failure to open; "r"; Open a file for reading
46  case in:
47  return mode | O_RDONLY;
48 //Destroy contents; Create new; "w"; Create a file for writing
49  case out:
50  return mode | O_WRONLY | O_CREAT | O_TRUNC;
51 //Append to file; Create new; "a"; Append to a file
52  case app:
53  case out|app:
54  return mode | O_WRONLY | O_CREAT | O_APPEND;
55 //Read from start; Error; "r+"; Open a file for read/write
56  case out|in:
57  return mode | O_RDWR;
58 //Write to end; Create new; "a+"; Open a file for read/write
59  case out|in|app:
60  case in|app:
61  return mode | O_RDWR | O_CREAT | O_APPEND;
62 //Destroy contents; Error; "wx"; Create a file for writing
63  default:
64  throw std::runtime_error("unknown posix file openmode");
65  }
66 }

68 struct posix_file_openmode
69 {
70  static int constexpr mode = calculate_posix_open_mode(om);
71 };
